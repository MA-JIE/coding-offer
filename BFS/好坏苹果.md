一个m*n的矩阵，里面每个数字代表苹果的好坏，0表示没有放苹果，1表示好苹果，2表示坏苹果，每一天坏苹果周围上下左右的好苹果都会变成坏苹果，问最多几天所有的好苹果会变成坏苹果，如果最后还有好苹果返回-1 <br>

``` cpp
vector<vector<int>> updateMatrix(vector<vector<int>>& mat) {
    int m = mat.size(), n = mat[0].size();
    vector<vector<int>> dirs{{0,-1},{-1,0},{0,1},{1,0}};
    queue<pair<int, int>> q;
    for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
            if (mat[i][j] == 0) q.push({i, j});
            else mat[i][j] = -1;
        }
    }
    while (!q.empty()) {
        auto t = q.front(); q.pop();
        for (auto dir : dirs) {
            int x = t.first + dir[0], y = t.second + dir[1];
            if (x < 0 || x >= m || y < 0 || y >= n || mat[x][y] != -1) continue;
            mat[x][y] = mat[t.first][t.second] + 1;
            q.push({x, y});
        }
    }
    return mat;
}
这段代码首先初始化了一个队列和四个方向，然后遍历整个矩阵，将所有的0添加进队列，同时将1的位置设置为-1，表示未访问过。然后开始进行BFS，每次从队列中取出一个元素，然后搜索其上下左右四个方向。如果在范围内且未访问过，那么就将其距离设置为当前元素的距离+1，并将其加入队列。最后返回修改后的矩阵。

这是一个典型的广度优先搜索（BFS）问题，我们可以使用队列来解决。

首先，我们遍历整个矩阵，将所有的坏苹果（2）添加进队列。

然后，我们开始进行BFS，每次从队列中取出一个元素，然后搜索其上下左右四个方向。如果在范围内且是好苹果（1），那么就将其变为坏苹果，并将其加入队列。

最后，我们再次遍历整个矩阵，如果还有好苹果，那么就返回-1，否则返回天数。

以下是C++代码实现：

int orangesRotting(vector<vector<int>>& grid) {
    int m = grid.size(), n = grid[0].size();
    vector<vector<int>> dirs{{0,-1},{-1,0},{0,1},{1,0}};
    queue<pair<int, int>> q;
    int fresh = 0, days = -1;
    for (int i = 0; i < m; ++i) {
        for (int j = 0; j < n; ++j) {
            if (grid[i][j] == 2) q.push({i, j});
            else if (grid[i][j] == 1) fresh++;
        }
    }
    while (!q.empty()) {
        int sz = q.size();
        for (int i = 0; i < sz; ++i) {
            auto t = q.front(); q.pop();
            for (auto dir : dirs) {
                int x = t.first + dir[0], y = t.second + dir[1];
                if (x < 0 || x >= m || y < 0 || y >= n || grid[x][y] != 1) continue;
                grid[x][y] = 2;
                q.push({x, y});
                fresh--;
            }
        }
        days++;
    }
    return fresh == 0 ? max(0, days) : -1;
}
```
