# 二叉树的层序遍历
给你一个二叉树，请你返回其按 层序遍历 得到的节点值。 （即逐层地，从左到右访问所有节点）。<br>
示例:<br>
二叉树：[3,9,20,null,null,15,7],<br>
    3  <br>
   / \ <br>
  9  20 <br>
    /  \ <br>
   15   7<br>
返回其层次遍历结果：<br>
[<br>
  [3],<br>
  [9,20],<br>
  [15,7] <br>
] <br>
思路:借助队列来完成层序遍历，以上述示例为例:<br>
tmp[3] --> res[[3]] --> tmp[9,20] --> res[[3],[9,20]] --> tmp[15,7] --> res[[3],[9,20],[15,7]] 
``` cpp
class Solution
{
public:
    vector<vector<int>> levelOrder(TreeNode *root)
    {
        vector<vector<int>> res;
        //空指针返回
        if (root == nullptr)
        {
            return res;
        }
        //构建队列存储节点
        queue<TreeNode *> tmp;
        tmp.push(root);
        while (!tmp.empty())
        {
            //存放每一层节点的个数
            int currentsize = tmp.size();
            res.emplace_back(vector<int>());
            for (int i = 0; i < currentsize; i++)
            {
                //依次取出队列中的首元素
                TreeNode *node = tmp.front();
                tmp.pop();
                //将值加入res中
                res.back().emplace_back(node->val);
                //向队列中加入下一层节点
                if (node->left)
                {
                    tmp.push(node->left);
                }
                if (node->right)
                {
                    tmp.push(node->right);
                }
            }
        }
        return res;
    }
};
```
